class Solution{
public:
    int minimumDays(int S, int N, int M){
        // code here
        int totalFoodRequired = S*M;
        int sundays = S/7;
        int shopOpenDays = S - sundays;
        
        int ans = totalFoodRequired/N + (totalFoodRequired%N ? 1 : 0);
        return ans > shopOpenDays ? -1 : ans;
    }
};

    }
